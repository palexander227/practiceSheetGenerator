






 If $x=2y+1$, what is $4y$ in terms of $x$?


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item   $x-1$
		\item  $2x-1$
		\item  $2x+2$
		\item  $2x-2$%
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item   $x-1$
		\item  $2x-1$
		\item  $2x+2$
		\addtocounter{enumii}{1}
		\item  $2x-2$%
		\item  $2x-4$
	\end{enumerate}
\else
\fi

\ifactodd
