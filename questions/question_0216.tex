



  When is $\mathlarger{\frac{\sqrt{x-4}}{x-4}}$ undefined?\\


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item   $x = 4$
		\item  $x < 4$
		\item   $x \leq 4$%
		\item  $x \geq 4$
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item   $x = 4$
		\item  $x < 4$
		\item   $x \leq 4$%
		\addtocounter{enumii}{1}
		\item  $x \geq 4$
		\item  $x = 0$
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item   $x = 4$
		\item  $x < 4$
		\item   $x \leq 4$%
		\item  $x \geq 4$
		\item  $x = 0$
	\end{enumerate}
\else
\fi
