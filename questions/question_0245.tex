


 Suppose $a$ and $b$ are both non-zero rational numbers.  Which must always be rational?


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item  $ab$
		\item  $\frac{a}{b}$
		\item  $\frac{b}{a}$
		\item  All of the above %
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item   $a+b$
		\item  $ab$
		\item  $\frac{a}{b}$
		\addtocounter{enumii}{1}
		\item  $\frac{b}{a}$
		\item  All of the above %
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item   $a+b$
		\item  $ab$
		\item  $\frac{a}{b}$
		\item  $\frac{b}{a}$
		\item  All of the above %
	\end{enumerate}
\else
\fi

\ifgridin
  All of the above %

\fi
