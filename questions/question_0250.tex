


 Which of the following can be the first three terms of an arithmetic sequence?


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item   $1,5,8$
		\item  $2,4,8$
		\item  $-1,1,-1$
		\item  $-2,1,4$%
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item   $1,5,8$
		\item  $2,4,8$
		\item  $-1,1,-1$
		\addtocounter{enumii}{1}
		\item  $-2,1,3$
		\item  $-2,1,4$%
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item   $1,5,8$
		\item  $2,4,8$
		\item  $-1,1,-1$
		\item  $-2,1,3$
		\item  $-2,1,4$%
	\end{enumerate}
\else
\fi
