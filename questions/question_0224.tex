



  Which of the following functions satisfy $f(-x)=f(x)$ for all real numbers $x$?


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item $x^{2}+x+1$
		\item   $x^{2}+2|x|$%
		\item  $|x-1|$
		\item   $x^{3}+2$
	\end{enumerate}

\fi

		\item   $x^{2}+2|x|$%

		\item   $x^{3}+2$
	\end{enumerate}

\fi
