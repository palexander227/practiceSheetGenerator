


 If $y=x^{3}$ and $z^{3}=y^{2}$, what is $z$ in terms of $x$?


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item   $x$
		\item  $\sqrt[3]{x}$%
		\item  $x^{2}$
		\item  $x^{3}$
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item   $x$
		\item  $\sqrt[3]{x}$%
		\item  $x^{2}$
		\addtocounter{enumii}{1}
		\item  $x^{3}$
		\item  $2x$
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item   $x$
		\item  $\sqrt[3]{x}$%
		\item  $x^{2}$
		\item  $x^{3}$
		\item  $2x$
	\end{enumerate}
\else
\fi
