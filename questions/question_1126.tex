 
If {\Large$\frac{x}{4}$}$=4-\sqrt{x-4}$,
what are the possible values for $x$?


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item $x\in\{8 \} $ % 
		\item $x\in\{40 \} $ 
		\item $x\in\{8,40 \} $
		\item $x\in\{24\pm8\sqrt{6} \}$
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item $x\in \O $ 
		\item $x\in\{8 \} $ % 
		\item $x\in\{40 \} $

		\item $x\in\{8,40 \} $
		\item $x\in\{24\pm8\sqrt{6} \}$
	\end{enumerate}
\else
\fi
