


 If for some integer $n>2$, \[2^{n+1}-2^{n}=k-1\] then what is $2^{n-1}$ in terms of $k$?\\




\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item   $\frac{k-1}{2}$%
		\item  $k$
		\item  $2k$
		\item   $2k-2$
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item   $\frac{k-1}{2}$%
		\item  $k-1$
		\item  $k$
		\addtocounter{enumii}{1}
		\item  $2k$
		\item   $2k-2$
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item   $\frac{k-1}{2}$%
		\item  $k-1$
		\item  $k$
		\item  $2k$
		\item   $2k-2$
	\end{enumerate}
\else
\fi

\ifgridin
   $\frac{k-1}{2}$%
		
\else
\fi
