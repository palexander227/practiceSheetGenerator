







 Suppose $(3-2i)x$ is a real number.  Then what is a possible value of $x$?


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item   $2+3i$
		\item  $2-3i$
		\item  $3-2i$
		\item  $3+2i$%
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item   $2+3i$
		\item  $2-3i$
		\item  $3-2i$
		\addtocounter{enumii}{1}
		\item  $3+2i$%
		\item   $11$
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item   $2+3i$
		\item  $2-3i$
		\item  $3-2i$
		\item  $3+2i$%
		\item   $11$
	\end{enumerate}
\else
\fi
