




  For any integer $n$, which of the following is always 1 more than double an even number?



\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item  $n+1$ 
		\item  $2n+1$  
		\item  $4n+1$ % 
		\item  $4n+3$
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item  $n+1$ 
		\item  $2n+1$  
		\item  $4n+1$ % 
		\addtocounter{enumii}{1}
		\item  $4n+3$
		\item  $4n$
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item  $n+1$ 
		\item  $2n+1$  
		\item  $4n+1$ % 
		\item  $4n+3$
		\item  $4n$
	\end{enumerate}
\else
\fi

\ifgridin
  $4n+1$ % 
		
\else
\fi
