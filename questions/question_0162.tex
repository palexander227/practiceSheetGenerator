\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item  $x^{2}-1$
		\item  $x^{2}-5x+6$
		\item  $x^{2}-2x+1$%
		\item  $x^{2}+6x+8$
	\end{enumerate}
\else
\fi

	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]

		\item  $x^{2}-2x+1$%
		\addtocounter{enumii}{1}

		\item  $x^{2}-9x+20$
	\end{enumerate}

\fi
