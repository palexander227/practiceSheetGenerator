






 Suppose $A$ is an $a\times b$ matrix and $B$ is a $c\times d$ matrix and the product $AB$ exists.  Which of the following must be true?


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item   $a=b$
		\item  $b=c$%
		\item  $c=d$
		\item  $b=d$
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item   $a=b$
		\item  $b=c$%
		\item  $c=d$
		\addtocounter{enumii}{1}
		\item  $a=c$
		\item  $b=d$
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item   $a=b$
		\item  $b=c$%
		\item  $c=d$
		\item  $a=c$
		\item  $b=d$
	\end{enumerate}
\else
\fi

  $b=c$%

\fi
