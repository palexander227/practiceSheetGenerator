


  Consider the following system of equations:
\[\begin{array}{lcl}
6x + 3y &=& -1\\
mx + 9y &=& -4
\end{array}
\]
For what value of $m$ does the above system have no solutions?


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item   $6$ 
		\item  $8$
		\item  $16$
		\item  $18$%
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item   $6$ 
		\item  $8$
		\item  $10$
		\addtocounter{enumii}{1}
		\item  $16$
		\item  $18$%
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item   $6$ 
		\item  $8$
		\item  $10$
		\item  $16$
		\item  $18$%
	\end{enumerate}
\else
\fi

\ifgridin
  $18$%

\else
\fi
