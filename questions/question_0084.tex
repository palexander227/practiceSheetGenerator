



  If for some integer $k\geq 3$, \[3^{k+1}+3^{k}=2^{s}\] then what is $3^{k+2}$ in terms of $s$?\\



\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item   $2^{s}$
		\item  $9\cdot 2^{s}$
		\item  $6\cdot2^{s-2}$
		\item   $9\cdot2^{s-2}$%
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item   $2^{s}$
		\item  $9\cdot 2^{s}$
		\item  $3\cdot2^{s-2}$
		\addtocounter{enumii}{1}
		\item  $6\cdot2^{s-2}$
		\item   $9\cdot2^{s-2}$%
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item   $2^{s}$
		\item  $9\cdot 2^{s}$
		\item  $3\cdot2^{s-2}$
		\item  $6\cdot2^{s-2}$
		\item   $9\cdot2^{s-2}$%
	\end{enumerate}
\else
\fi

\ifgridin
   $9\cdot2^{s-2}$%

\else
\fi
