


 Which of the following expressions is the same as $\frac{n+1!}{n!}$ for any integer $n$?


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item  $n$
		\item  $n!$
		\item  $n+1$%
		\item  $n+1!$
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item   1
		\item  $n$
		\item  $n!$
		\addtocounter{enumii}{1}
		\item  $n+1$%
		\item  $n+1!$
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item   1
		\item  $n$
		\item  $n!$
		\item  $n+1$%
		\item  $n+1!$
	\end{enumerate}
\else
\fi

\ifgridin
  $n+1$%
		
\else
\fi
