Suppose $m$ and $n$ are positive integers and $m+n=mn$. Which of the following could be true?

\hspace{3cm}I. $n-1$ is odd.

\hspace{3cm}II. $m$ and $n$ are odd.

\hspace{3cm}III. $m$ and $n$ are even.



\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item I and II.
		\item I and III.%
		\item II only.
		\item III only.
 
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item I and II.
		\item I and III.%
		\item I only.
		\addtocounter{enumii}{1}
		\item II only.
		\item III only.
 
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item I and II.
		\item I and III.%
		\item I only.
		\item II only.
		\item III only.
 
	\end{enumerate}
\else
\fi
