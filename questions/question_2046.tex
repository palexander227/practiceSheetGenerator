



  Which of the following formulas defines an odd function of the variable $x$?


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item    $x+x^{3}$%
		\item $x^{2}+x+1$
		\item   $x^{2}+2|x|$
		\item   $x^{3}+2$
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item    $x+x^{3}$%
		\item $x^{2}+x+1$
		\item   $x^{2}+2|x|$
		\addtocounter{enumii}{1}
		\item  $|x-1|$
		\item   $x^{3}+2$
	\end{enumerate}
\else
\fi

\ifactodd
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\item    $x+x^{3}$%
		\item $x^{2}+x+1$
		\item   $x^{2}+2|x|$
		\item  $|x-1|$
		\item   $x^{3}+2$
	\end{enumerate}
\else
\fi
