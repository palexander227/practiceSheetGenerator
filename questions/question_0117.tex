

  Suppose $x^{2}-c^{2}$ has 7 as a root.  Then what is the other root? 


\ifsat
	\begin{enumerate}[label=\Alph*)]
		\item   $-14$
		\item  $-7$%
		\item  $-49$
		\item  $49$
	\end{enumerate}
\else
\fi

\ifacteven
	\begin{enumerate}[label=\textbf{\Alph*.},itemsep=\fill,align=left]
		\setcounter{enumii}{5}
		\item   $-14$
		\item  $-7$%
		\item  $-49$
		\addtocounter{enumii}{1}
		\item  $7$
		\item  $49$
	\end{enumerate}
\else
\fi

		\item  $49$
	\end{enumerate}
\else
\fi
